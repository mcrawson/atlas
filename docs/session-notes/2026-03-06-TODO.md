### Integration Layer Base Class
Create the foundation for all platform integrations:

```python
# atlas/integrations/platforms/base.py

from abc import ABC, abstractmethod

class PlatformIntegration(ABC):
    name: str
    icon: str

    @abstractmethod
    def get_requirements(self) -> list:
        """What's needed to publish to this platform"""
        pass

    @abstractmethod
    def validate(self, product) -> bool:
        """Check if product meets platform requirements"""
        pass

    @abstractmethod
    async def publish(self, product) -> dict:
        """Publish to the platform"""
        pass

    @abstractmethod
    async def check_status(self, submission_id) -> dict:
        """Check publishing status"""
        pass
```
